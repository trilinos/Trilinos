  TRIBITS_ADD_EXECUTABLE(
    SolverFactory_Tpetra_Galeri_Ex

    SolverFactory_Tpetra_Galeri_Ex

    TRIBITS_ADD_TEST(
      SolverFactory_Tpetra_Galeri_Ex

      SolverFactory_Tpetra_Galeri_Ex
      NAME PerfTest_BlockCG

      ARGS "--verbose --solverName=\"Block CG\" --matrixType=Laplace3D --nx=20 --ny=20 --nz=20 --max-iters=100"
