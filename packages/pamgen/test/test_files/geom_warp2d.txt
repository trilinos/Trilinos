     double r = sqrt(inxcoord*inxcoord+inycoord*inycoord);
     double theta = atan2(inycoord,inxcoord);
     if(r > 0.5) 
      {
        theta = theta + (3.14159 / 4.0)*((r-0.5)/0.5);
        outxcoord = r*cos(theta);
	outycoord = r*sin(theta);
      }
     "
