  ENDIF()

  ASSERT_DEFINED(${PACKAGE_NAME}_ENABLE_ML)

  ENDIF()

  IF (${PACKAGE_NAME}_ENABLE_Amesos)
    ADD_SUBDIRECTORY(amesos)
  ENDIF()

  ASSERT_DEFINED(${PACKAGE_NAME}_ENABLE_AztecOO)

  ENDIF()
ENDIF()
