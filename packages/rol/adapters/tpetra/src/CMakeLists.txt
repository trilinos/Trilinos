IF (ROL_ENABLE_Tpetra) 

  TRIBITS_SET_AND_INC_DIRS( DIR ${CMAKE_CURRENT_SOURCE_DIR}/vector )
  TRIBITS_INCLUDE_DIRECTORIES( ${DIR} )
  APPEND_GLOB(HEADERS ${DIR}/*.hpp)
  
  TRIBITS_SET_AND_INC_DIRS( DIR ${CMAKE_CURRENT_SOURCE_DIR}/function )
  TRIBITS_INCLUDE_DIRECTORIES( ${DIR} )
  APPEND_GLOB(HEADERS ${DIR}/*.hpp)
  
  IF (ROL_ENABLE_Teuchos) 
  
    TRIBITS_SET_AND_INC_DIRS( DIR ${CMAKE_CURRENT_SOURCE_DIR}/sol )
    TRIBITS_INCLUDE_DIRECTORIES( ${DIR} )
    APPEND_GLOB(HEADERS ${DIR}/*.hpp)
  
  ENDIF()

  # add in the MPI code
  IF( TPL_ENABLE_MPI ) 
    TRIBITS_SET_AND_INC_DIRS( DIR ${CMAKE_CURRENT_SOURCE_DIR}/mpi )
    TRIBITS_INCLUDE_DIRECTORIES( ${DIR} )
    APPEND_GLOB(HEADERS ${DIR}/*.hpp)
  ENDIF()

  IF (ROL_ENABLE_pebbl)
    TRIBITS_SET_AND_INC_DIRS( DIR ${CMAKE_CURRENT_SOURCE_DIR}/pebbl )
    TRIBITS_INCLUDE_DIRECTORIES( ${DIR} )
    APPEND_GLOB(HEADERS ${DIR}/*.hpp)
  ENDIF()
