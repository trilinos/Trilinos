00045     const Teuchos::ArrayView<typename TypeMap<Superlu,typename Matrix::scalar_type>::type>& nzval,
00046     const Teuchos::ArrayView<int>& colind, 
00047     const Teuchos::ArrayView<int>& rowptr, 
00048     const Teuchos::Ptr<SLU::SuperMatrix>& A 
00049     )                         
00050     {
00051       typedef typename Matrix::scalar_type                     scalar_type;
00052       typedef typename Matrix::global_ordinal_type                 go_type;
00053       typedef typename Matrix::global_size_type                    gs_type;
00054       typedef typename TypeMap<Amesos::Superlu,scalar_type>::type slu_type;
00055       // Get the SLU data type for this type of matrix
00056       SLU::Dtype_t dtype = TypeMap<Amesos::Superlu,scalar_type>::dtype;
00057 
00058       // Extract the necessary information from mat and call SLU function
00059       using Teuchos::Array;
00060       using Teuchos::ArrayView;
00061 
00062       int nnz, rows, cols;
00063       nnz  = Teuchos::as<int>(mat->getGlobalNNZ());
00064       rows = Teuchos::as<int>(mat->getGlobalNumRows());
00065       cols = Teuchos::as<int>(mat->getGlobalNumCols());
00066 
00067       TEST_FOR_EXCEPTION( Teuchos::as<int>(nzval.size()) < nnz,
00068         std::runtime_error,
00069         "nzval array not large enough to hold data");
00070       TEST_FOR_EXCEPTION( Teuchos::as<int>(colind.size()) < nnz,
00071         std::runtime_error,
00072         "colind array not large enough to hold data");
00073       TEST_FOR_EXCEPTION( Teuchos::as<int>(rowptr.size()) < rows + 1,
00074         std::runtime_error,
00075         "rowptr array not large enough to hold data");
00076       
00077       Array<scalar_type> nzval_tmp(nzval.size());
00078       Array<go_type> colind_tmp(colind.size());
00079       Array<gs_type> rowptr_tmp(rowptr.size());
00080       size_t nnz_ret = 0;
00081 
00082       // Actually uses the compressed-row-store format, and tells Superlu this
00083       // while creating a compressed-column store.
00084       mat->getCrs((ArrayView<scalar_type>)nzval_tmp,
00085         (ArrayView<go_type>)colind_tmp, (ArrayView<gs_type>)rowptr_tmp, nnz_ret);
00086 
00087       TEST_FOR_EXCEPTION( nnz_ret != Teuchos::as<size_t>(nnz),
00088         std::runtime_error,
00089         "Number of nonzeros returned by getCrs() different from getGlobalNNZ()");
00090       
00091       /* Convert types
00092        *
00093        * Note: We cannot simply convert when necessary.  That is, we cannot
00094        * check whether the matrix types are double, int, and int, and then do
00095        * a simple ArrayView::assign if they are the same, since the compiler
00096        * still has to do the static linking and cannot find an appropriate
00097        * function call in the case that the types are not double, int, and int
00098        */
00099       {
00100         typename Array<scalar_type>::size_type i, size;
00101         size = nzval_tmp.size();
00102         for ( i = 0; i < size; ++i ){
00103           nzval[i] = Teuchos::as<slu_type>(nzval_tmp[i]);
00104         }
00105       }
00106       {
00107         typename Array<go_type>::size_type i, size;
00108         size = colind_tmp.size();
00109         for ( i = 0; i < size; ++i ){
00110           colind[i] = Teuchos::as<int>(colind_tmp[i]);
00111         }
00112       }
00113       {
00114         typename Array<gs_type>::size_type i, size;
00115         size = rowptr_tmp.size();
00116         for ( i = 0; i < size; ++i ){
00117           rowptr[i] = Teuchos::as<int>(rowptr_tmp[i]);
00118         }
00119       }
00120       // end conversions
00121 
00122       FunctionMap<Superlu,scalar_type>::create_CompRow_Matrix(
00123         A.getRawPtr(), rows, cols, nnz, nzval.getRawPtr(),
00124         colind.getRawPtr(), rowptr.getRawPtr(), SLU::SLU_NR,
00125         dtype, SLU::SLU_GE);
