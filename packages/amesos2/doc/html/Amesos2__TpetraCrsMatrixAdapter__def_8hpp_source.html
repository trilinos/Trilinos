00354 /** \brief Gets a compressed-row storage summary of \c this
00355  *
00356  * Extracts a compressed-row storage format of the matrix and stores the
00357  * information in the user-supplied containers.
00358  *
00359  * \param [out] nzval will hold the values of the nonzero entries of \c this
00360  * \param [out] colind will hold the column indices of \c this for each row.
00361  * \param [out] rowptr is of size <tt>nrow + 1</tt> and \c rowptr[j] stores
00362  *              the location in \c nzval and \c colind which starts row \c j
00363  *              of \c this.  <tt>rowptr[nrow] = nnz</tt>, where \c nrow is
00364  *              the number of rows in this matrix.
00365  * \param [out] nnz is the number of nonzero entries in this matrix.
00366  * \param [in]  local If \c false, the processor with ID \c root will contain
00367  *              a representation of the global matrix.  If \c true, then each
00368  *              processor will end up with a CRS representation of the matrix
00369  *              rows that it owns.
00370  * \param [in]  root Is the processor ID of the node that will end up with
00371  *              the CRS representation of the matrix.
00372  *
00373  * \exception std::runtime_error Thrown if \c nzval or \c colind is not
00374  * large enough to hold the global number of nonzero values.
00375  *
00376  * \exception std::runtime_error Thrown if \c rowptr is not at least
00377  * <tt>nrow + 1</tt> in size, the required size.
00378  */
00379 template< typename Scalar, typename LocalOrdinal, typename GlobalOrdinal, class Node >
00380 void
00381 MatrixAdapter<
00382   CrsMatrix<
00383     Scalar,
00384     LocalOrdinal,
00385     GlobalOrdinal,
00386     Node
00387     >
00388   >::getCrs(
00389     const Teuchos::ArrayView<Scalar> nzval,
00390     const Teuchos::ArrayView<GlobalOrdinal> colind,
00391     const Teuchos::ArrayView<Tpetra::global_size_t> rowptr,
00392     size_t& nnz,
00393     bool local,
00394     int root)
00395 {
00396   Teuchos::RCP<const Teuchos::Comm<int> > comm = getComm();
00397   const int rank = comm->getRank();
00398 
00399   // Could we have numrows be LocalOrdinal or GlobalOrdinal depending on $local
00400   GlobalOrdinal numrows = 0;
00401   if ( local ){
00402     numrows = mat_->getNodeNumRows();
00403     nnz = mat_->getNodeNumEntries();
00404   } else {
00405     numrows = mat_->getGlobalNumRows();
00406     nnz = mat_->getGlobalNumEntries();
00407   }

00409       && (Teuchos::as<size_t>(nzval.size()) < nnz)) ||
00410     (local && (Teuchos::as<size_t>(nzval.size()) < nnz)),
00411     std::runtime_error,
00412     "nzval not large enough to hold nonzero values");
00413   TEST_FOR_EXCEPTION( ((rank == root)
00414       && (Teuchos::as<size_t>(colind.size()) < nnz)) ||
00415     (local && (Teuchos::as<size_t>(colind.size()) < nnz)),
00416     std::runtime_error,
00417     "colind not large enough to hold index values");
00418   TEST_FOR_EXCEPTION( ((rank == root)
00419       && (Teuchos::as<GlobalOrdinal>(rowptr.size()) < numrows + 1)) ||
00420     (local && (Teuchos::as<GlobalOrdinal>(rowptr.size()) < numrows + 1)),
00421     std::runtime_error,
00422     "rowptr not large enough to hold row index values");
00423   // TODO: Is there possibly a better way to test locality?
00424   if ( !mat_->getRowMap()->isDistributed() || local ){ // This matrix is *not*
00425     // distributed
00426     global_size_type rowInd = 0;
00427     for( GlobalOrdinal row = 0; row < numrows; ++row ){
00428       rowptr[row] = rowInd;
00429       GlobalOrdinal myRow = mat_->getRowMap()->getGlobalElement(row);
00430       size_t rowNNZ = mat_->getNumEntriesInLocalRow(myRow);
00431       size_t nnzRet = 0;
00432       mat_->getGlobalRowCopy(myRow,
00433         colind.view(rowInd,rowNNZ),
00434         nzval.view(rowInd,rowNNZ),
00435         nnzRet);
00436       TEST_FOR_EXCEPTION( rowNNZ != nnzRet,
00437         std::runtime_error,
00438         "Number of values returned different from \
00439                           number of values reported");
00440       rowInd += rowNNZ;
00441     }
00442     rowptr[numrows] = nnz;
00443   } else {                      // This matrix is distributed
00444     using Teuchos::Array;
00445     using Teuchos::Ptr;
00446     using Teuchos::OrdinalTraits;
00447 
00448     /* Each processor has an array the length of globalNumRows, each index
00449      * initialized to the max ordinal.  For each global index that the
00450      * processor owns, they write their rank in that spot, however the
00451      * processor that has rank == root write a -1.  We do a global array MIN
00452      * reduction on those arrays.  The root writes -1 because we want to
00453      * assure that we do local operations as much as we can.  It may be
00454      * possible that two nodes have a copy of the same row, but we want to
00455      * prefer the root's copy in all cases.
00456      * 
00457      * Now the processor with rank == root goes through the globally reduced
00458      * array: For indices that have -1, then it does a local row copy, for
00459      * others it gets the row copy from that processor indicated.  Meanwhile,
00460      * other processors are sending the rows they've been assigned (the global
00461      * rows corresponding to the indices in the reduced array that == their
00462      * rank) to the root.
00463      */

00466     Array<int> myRows(numrows,OrdinalTraits<int>::max());

00468       // if global index is found on this node
00469       if( mat_->getRowMap()->isNodeGlobalElement(row) ){
00470         myRows[row] = (rank == root) ? -1 : rank;
00471       }
00472     }
00473     // now perform global reduce of these arrays
00474     Teuchos::MinValueReductionOp<int,int> op;
00475     Teuchos::reduceAll(*comm, op, (int)myRows.size(),
00476       myRows.getRawPtr(), minOwner.getRawPtr());
00477 
00478     if( rank == root ){
00479       global_size_type rowInd = 0;
00480       size_t rowNNZ, nnzRet;
00481       for( GlobalOrdinal row = 0; row < numrows; ++row ){
00482         int from = minOwner[row];
00483         if( from == -1 ){       // Copy from myself
00484           rowptr[row] = rowInd;
00485           rowNNZ = mat_->getNumEntriesInGlobalRow(row);
00486           nnzRet = 0;
00487           mat_->getGlobalRowCopy(row,
00488             colind.view(rowInd,rowNNZ),
00489             nzval.view(rowInd,rowNNZ),
00490             nnzRet);
00491           TEST_FOR_EXCEPTION( rowNNZ != nnzRet,
00492             std::runtime_error,
00493             "Number of values returned different from number of values reported");
00494           rowInd += rowNNZ;
00495         } else {
00496           rowptr[row] = rowInd;
00497           Teuchos::receive(*comm, from, &rowNNZ);
00498           Teuchos::receive(*comm, from, (int)rowNNZ,
00499             colind.view(rowInd, rowNNZ).getRawPtr());
00500           Teuchos::receive(*comm, from, (int)rowNNZ,
00501             nzval.view(rowInd, rowNNZ).getRawPtr());
00502           rowInd += rowNNZ;
00503         }
00504         rowptr[numrows] = nnz;
00505       }
00506     } else {                  // I am not the root
00507       size_t localNNZ = mat_->getNodeNumEntries();
00508       Teuchos::Array<GlobalOrdinal> colind_l(colind);
00509       Teuchos::Array<Scalar> nzval_l(nzval);
00510       colind_l.resize(localNNZ);
00511       nzval_l.resize(localNNZ);
00512       GlobalOrdinal rowInd = 0;
00513       size_t rowNNZ, nnzRet;
00514       for( GlobalOrdinal row = 0; row < numrows; ++row ){
00515         int from = minOwner[row];
00516         if( from == rank ){   // I am responsible for this row
00517           rowNNZ = mat_->getNumEntriesInGlobalRow(row);
00518           nnzRet = 0;
00519           mat_->getGlobalRowCopy(row,
00520             colind_l.view(rowInd, rowNNZ),
00521             nzval_l.view(rowInd, rowNNZ),
00522             nnzRet);
00523           TEST_FOR_EXCEPTION( rowNNZ != nnzRet,
00524             std::runtime_error,
00525             "Number of values returned different from number of values reported");

00527           Teuchos::send(*comm, rowNNZ, root);
00528           Teuchos::send(*comm, (int)rowNNZ,

00531             nzval_l.view(rowInd,rowNNZ).getRawPtr(), root);
00532           rowInd += rowNNZ;   // This is purely local to this node
00533         }
00534       }
00535     }

00585 template< typename Scalar, typename LocalOrdinal, typename GlobalOrdinal, class Node >

00587 MatrixAdapter<
00588   CrsMatrix<
00589     Scalar,
00590     LocalOrdinal,
00591     GlobalOrdinal,
00592     Node
00593     >
00594   >::getCcs(
00595     const Teuchos::ArrayView<Scalar> nzval,
00596     const Teuchos::ArrayView<GlobalOrdinal> rowind,
00597     const Teuchos::ArrayView<Tpetra::global_size_t> colptr,
00598     size_t& nnz,
00599     bool local,
00600     int root)
00601 {
00602   using Teuchos::Array;
00603   using Teuchos::ArrayView;
00604 
00605   Teuchos::RCP<const Teuchos::Comm<int> > comm = getComm();
00606   const int rank = comm->getRank();
00607 
00608   global_size_type numCols = 0;
00609   global_size_type numRows = 0;
00610   
00611   if ( local ){
00612     numCols = mat_->getNodeNumCols();
00613     numRows = mat_->getNodeNumRows();
00614   } else {
00615     numCols = mat_->getGlobalNumCols();
00616     numRows = mat_->getGlobalNumRows();
00617   }
00618   
00619   Array<scalar_type> nzval_tmp(nzval.size());
00620   Array<GlobalOrdinal> colind(nzval.size());
00621   Array<global_size_type> rowptr(numRows + 1);
00622 
00623   this->getCrs(nzval_tmp, colind, rowptr, nnz, local, root);
00624   /* We now have a compressed-row storage format of this matrix.  We transform
00625    * this into a compressed-column format using a distribution-counting sort
00626    * algorithm, which is described by D.E. Knuth in TAOCP Vol 3, 2nd ed pg 78.
00627    
00628    * Check the conditional execution here.  Also, we were having trouble with
00629    * Tpetra::CrsMatrix not reporting the correct global number of columns.
00630    */
00631   if( local || rank == root ){
00632     // Count the number of entries in each column
00633     Array<global_size_type> count(numCols + 1, 0);
00634     typename Array<GlobalOrdinal>::iterator ind_it;
00635     for( ind_it = colind.begin(); ind_it != colind.end(); ++ind_it ){
00636       ++(count[(*ind_it) + 1]);
00637     }
00638 
00639     // Accumulate
00640     typename Array<global_size_type>::iterator cnt_it;
00641     for( cnt_it = count.begin() + 1; cnt_it != count.end(); ++cnt_it ){
00642       *cnt_it = *cnt_it + *(cnt_it - 1);
00643     }
00644     // This becomes the array of column pointers
00645     colptr.assign(count);
00646 
00647     /* Move the nonzero values into their final place in nzval, based on the
00648      * counts found previously.
00649      *
00650      * This sequence deviates from Knuth's algorithm a bit, following more
00651      * closely the description presented in Gustavson, Fred G. "Two Fast
00652      * Algorithms for Sparse Matrices: Multiplication and Permuted
00653      * Transposition" ACM Trans. Math. Softw. volume 4, number 3, 1978, pages
00654      * 250--269, http://doi.acm.org/10.1145/355791.355796.
00655      *
00656      * The row indices end up in sorted order
00657      */
00658 
00659     for( global_size_type row = 0; row < numRows; ++row ){
00660       GlobalOrdinal u = rowptr[row];
00661       GlobalOrdinal v = rowptr[row + 1];
00662       for( GlobalOrdinal j = u; j < v; ++j ){
00663         GlobalOrdinal i = count[colind[j]];
00664         nzval[i] = nzval_tmp[j];
00665         rowind[i] = row;
