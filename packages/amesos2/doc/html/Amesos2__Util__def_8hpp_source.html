00084 template< typename Matrix>
00085 void Amesos::Util::setMaxProcesses(
00086   const Teuchos::RCP<Matrix>& A,
00087   int& maxProcesses)
00088 {
00089   int maxProcs = A->getComm()->getSize();
00090 
00091   switch(maxProcesses){
00092     case -3:
00093       maxProcesses = maxProcs; break;
00094     case -2:
00095       maxProcesses = (int) sqrt((double)maxProcs); break;
00096     case -1:      // We should do some testing on this
00097       // heuristic
00098       maxProcesses =
00099         1 + TEUCHOS_MAX(A.getGlobalNumRows() / 10000,
00100           A.getGlobalNumEntries() / 1000000);
00101       break;
00102   }
00103 
00104   if(maxProcesses <= 0) maxProcesses = 1;
00105   if(maxProcesses > maxProcs) maxProcesses = maxProcs;
00106 
00107   return;
00108 }
