
INCLUDE(PackageLibraryMacros)

SET(SOURCES "")

  NOX_LAPACK.H
  NOX_LAPACK_Matrix.H
  NOX_LAPACK_Vector.H
  NOX_LAPACK_LinearSolver.H
  NOX_LAPACK_Group.H
  NOX_LAPACK_Interface.H
  )

APPEND_SET(SOURCES
  NOX_LAPACK_Vector.C
  NOX_LAPACK_Group.C 
  )

IF(NOX_ENABLE_ABSTRACT_IMPLEMENTATION_LAPACK)
  PACKAGE_ADD_LIBRARY(
    noxlapack
    HEADERS ${HEADERS}
    SOURCES ${SOURCES}
    DEPLIBS nox
    )
ENDIF()
