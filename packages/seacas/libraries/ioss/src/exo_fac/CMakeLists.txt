INCLUDE(TribitsLibraryMacros)

if(${CMAKE_PROJECT_NAME} STREQUAL "SEACAS")
  TRIBITS_SUBPACKAGE(Ioexo_fac)
endif()
SET(HEADERS "")
SET(SOURCES "")

SET_AND_INC_DIRS(DIR ${CMAKE_CURRENT_SOURCE_DIR})
APPEND_GLOB(HEADERS ${DIR}/*.h)
APPEND_GLOB(SOURCES ${DIR}/*.C)

INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}/../"
  "${CMAKE_CURRENT_BINARY_DIR}/../"
)

SET(IOSS_DEPENDENCIES "")
if(${CMAKE_PROJECT_NAME} STREQUAL "SEACAS")
  IF (TPL_ENABLE_MPI)
    APPEND_SET(IOSS_DEPENDENCIES Iopx)
  ELSE()
    ADD_DEFINITIONS(-DNO_DOF_EXODUS_SUPPORT)
  ENDIF()
else()
  IF (SEACASExodus_ENABLE_MPI)
    APPEND_SET(IOSS_DEPENDENCIES Iopx)
  ELSE()
    ADD_DEFINITIONS(-DNO_DOF_EXODUS_SUPPORT)
  ENDIF()
endif()

TRIBITS_ADD_LIBRARY(
	Ioexo_fac
	HEADERS	${HEADERS}
	SOURCES ${SOURCES}
	DEPLIBS ${IOSS_DEPENDENCIES}
)

if(${CMAKE_PROJECT_NAME} STREQUAL "SEACAS")
  TRIBITS_SUBPACKAGE_POSTPROCESS()
endif()
