
SET(HEADERS "")
SET(SOURCES "")

IF (Trilinos_ENABLE_Zoltan)
   SET(ZOLTAN_LIB "zoltan")   
ELSE()
   ADD_DEFINITIONS(-DNO_ZOLTAN_SUPPORT)
ENDIF()

IF (Trilinos_ENABLE_ParMETIS)
   SET(PARMETIS_LIB "parmetis")
ELSE()
   ADD_DEFINITIONS(-DNO_PARMETIS_SUPPORT)
ENDIF()

SET_AND_INC_DIRS(DIR ${CMAKE_CURRENT_SOURCE_DIR})
APPEND_GLOB(HEADERS ${DIR}/*.h)
APPEND_GLOB(SOURCES ${DIR}/*.C)

INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}/../"
  "${CMAKE_CURRENT_BINARY_DIR}/../"
  "${Zoltan_INCLUDE_DIRS}"
)


TRIBITS_ADD_LIBRARY(
	Iopx
	HEADERS	${HEADERS}
	SOURCES ${SOURCES}
	DEPLIBS Ioss exodus ${ZOLTAN_LIB} ${PARMETIS_LIB})

IF (BUILD_TESTING)
  ENABLE_TESTING()
ENDIF (BUILD_TESTING)
