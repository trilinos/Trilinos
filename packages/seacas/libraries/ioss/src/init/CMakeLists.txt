INCLUDE(TribitsLibraryMacros)
INCLUDE(CheckIncludeFile)

if(${CMAKE_PROJECT_NAME} STREQUAL "SEACAS")
  TRIBITS_SUBPACKAGE(Ionit)
endif()
SET(HEADERS "")
SET(SOURCES "")

SET_AND_INC_DIRS(DIR ${CMAKE_CURRENT_SOURCE_DIR})
APPEND_GLOB(HEADERS ${DIR}/*.h)
APPEND_GLOB(SOURCES ${DIR}/*.C)

INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}/../"
  "${CMAKE_CURRENT_BINARY_DIR}/../"
)

if(${CMAKE_PROJECT_NAME} STREQUAL "SEACAS")
  SET(PREFIX "SEACAS")
  SET(TPL_PREFIX "TPL")
else()
  SET(PREFIX ${PACKAGE_NAME})
  SET(TPL_PREFIX ${PACKAGE_NAME})
endif()
SET(ENABLED ${SEACAS_PREFIX}Exodus)
ASSERT_DEFINED(${PREFIX}_ENABLE_${ENABLED})
IF (${PREFIX}_ENABLE_${ENABLED})
  IF (${${ENABLED}_ENABLE_MPI})
  ELSE()
    ADD_DEFINITIONS(-DNO_DOF_EXODUS_SUPPORT)
  ENDIF()
ELSE()
  ADD_DEFINITIONS(-DNO_EXODUS_SUPPORT)
ENDIF()

ASSERT_DEFINED(${TPL_PREFIX}_ENABLE_Pamgen)
IF (NOT ${${TPL_PREFIX}_ENABLE_Pamgen})
  ADD_DEFINITIONS(-DNO_PAMGEN_SUPPORT)
ENDIF()

IF (${${PREFIX}_ENABLE_ParaviewImesh})
ELSE()
  ADD_DEFINITIONS(-DNO_PARAVIEWIMESH_SUPPORT)
ENDIF()


TRIBITS_ADD_LIBRARY(
	Ionit
	HEADERS	${HEADERS}
	SOURCES ${SOURCES}
)

IF (BUILD_TESTING)
  ENABLE_TESTING()
ENDIF (BUILD_TESTING)

if(${CMAKE_PROJECT_NAME} STREQUAL "SEACAS")
  INSTALL(FILES ${HEADERS} DESTINATION
      ${CMAKE_INSTALL_PREFIX}/include/init/)
else()
  INSTALL(FILES ${HEADERS} DESTINATION
      ${CMAKE_INSTALL_PREFIX}/${Trilinos_INSTALL_INCLUDE_DIR}/init/)
endif()


if(${CMAKE_PROJECT_NAME} STREQUAL "SEACAS")
  TRIBITS_SUBPACKAGE_POSTPROCESS()
endif()
