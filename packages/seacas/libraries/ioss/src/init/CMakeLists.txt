INCLUDE(TribitsLibraryMacros)
INCLUDE(CheckIncludeFile)

SET(HEADERS "")
SET(SOURCES "")

SET_AND_INC_DIRS(DIR ${CMAKE_CURRENT_SOURCE_DIR})
APPEND_GLOB(HEADERS ${DIR}/*.h)
APPEND_GLOB(SOURCES ${DIR}/*.C)

INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}/../"
  "${CMAKE_CURRENT_BINARY_DIR}/../"
)

IF (${PACKAGE_NAME}_ENABLE_Pamgen)
SET(PAMGENLIB Iopg)
ELSE()
ADD_DEFINITIONS(-DNO_PAMGEN_SUPPORT)
ENDIF()

IF (${PACKAGE_NAME}_ENABLE_SEACASExodus)
  SET(EXODUSLIB Ioex)

  IF (${PACKAGE_NAME}_ENABLE_MPI AND ${PACKAGE_NAME}_ENABLE_Zoltan AND TPL_ENABLE_ParMETIS)
    SET(PAREXOLIB Iopx)
  ELSE()
    ADD_DEFINITIONS(-DNO_DOF_EXODUS_SUPPORT)
  ENDIF()
ELSE()
ADD_DEFINITIONS(-DNO_EXODUS_SUPPORT)
ENDIF()

IF (${PACKAGE_NAME}_ENABLE_ParaviewImesh)
SET(VISLIB Iovs)
ELSE()
ADD_DEFINITIONS(-DNO_PARAVIEWIMESH_SUPPORT)
ENDIF()


TRIBITS_ADD_LIBRARY(
	Ionit
	HEADERS	${HEADERS}
	SOURCES ${SOURCES}
	DEPLIBS Ioss ${EXODUSLIB} ${PAREXOLIB} Iogn Iohb Iotr ${PAMGENLIB} ${VISLIB}
)

IF (BUILD_TESTING)
  ENABLE_TESTING()
ENDIF (BUILD_TESTING)

INSTALL(FILES ${HEADERS} DESTINATION
    ${CMAKE_INSTALL_PREFIX}/${Trilinos_INSTALL_INCLUDE_DIR}/init/)
