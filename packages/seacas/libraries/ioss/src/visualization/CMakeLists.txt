
SET(HEADERS "")
SET(SOURCES "")

SET_AND_INC_DIRS(DIR ${CMAKE_CURRENT_SOURCE_DIR})
APPEND_GLOB(HEADERS ${DIR}/cgns/*.h)
APPEND_GLOB(SOURCES ${DIR}/cgns/*.C)
APPEND_GLOB(HEADERS ${DIR}/exodus/*.h)
APPEND_GLOB(SOURCES ${DIR}/exodus/*.C)
APPEND_GLOB(HEADERS ${DIR}/utils/*.h)
APPEND_GLOB(SOURCES ${DIR}/utils/*.C)

INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}/cgns/"
  "${CMAKE_CURRENT_SOURCE_DIR}/exodus/"
  "${CMAKE_CURRENT_SOURCE_DIR}/utils/"
  "${CMAKE_CURRENT_SOURCE_DIR}/../"
  "${CMAKE_CURRENT_BINARY_DIR}/../"
)

IF(${TPL_ENABLE_DLlib})
  ADD_DEFINITIONS(-DIOSS_DLOPEN_ENABLED)
ENDIF ()

TRIBITS_ADD_LIBRARY(
	Iovs
	HEADERS	${HEADERS}
	SOURCES ${SOURCES}
	DEPLIBS Ioss
)

IF (BUILD_TESTING)
  ENABLE_TESTING()
ENDIF (BUILD_TESTING)
