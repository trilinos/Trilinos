
SET(HEADERS "")
SET(SOURCES "")

SET_AND_INC_DIRS(DIR ${CMAKE_CURRENT_SOURCE_DIR})
APPEND_GLOB(HEADERS ${DIR}/ParaViewCatalystIossAdapter/include/*.h)
APPEND_GLOB(HEADERS ${DIR}/*.h)
APPEND_GLOB(SOURCES ${DIR}/*.C)

INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}/ParaViewCatalystIossAdapter/"
  "${CMAKE_CURRENT_SOURCE_DIR}/ParaViewCatalystIossAdapter/include/"
  "${CMAKE_CURRENT_SOURCE_DIR}/../"
  "${CMAKE_CURRENT_BINARY_DIR}/../"
)

IF(${TPL_ENABLE_DLlib})
  ADD_DEFINITIONS(-DIOSS_DLOPEN_ENABLED)
ENDIF ()

TRIBITS_ADD_LIBRARY(
	Iovs
	HEADERS	${HEADERS}
	SOURCES ${SOURCES}
	DEPLIBS Ioss
)

IF (BUILD_TESTING)
  ENABLE_TESTING()
ENDIF (BUILD_TESTING)
