TRIBITS_SUBPACKAGE(SuplibC)

SET(SUPLIBC_VERSION_MAJOR "3")
SET(SUPLIBC_VERSION_MINOR "08")
SET(SUPLIBC_VERSION_PATCH "0")
SET(SUPLIBC_VERSION "${SUPLIBC_VERSION_MAJOR}.${SUPLIBC_VERSION_MINOR}")
SET(SUPLIBC_VERSION_FULL "${SUPLIBC_VERSION}.${SUPLIBC_VERSION_PATCH}")

INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}"
  )

SET(SOURCES
  add_to_log.c
  getline.c
  adler.c
  copy_string.c
  term_width.c
  )

IF (CMAKE_HOST_SYSTEM_NAME STREQUAL "Windows")
   APPEND_GLOB(SOURCES XGetopt.c)
ENDIF()

TRIBITS_ADD_LIBRARY(
  suplib_c
  NOINSTALLHEADERS
  SOURCES ${SOURCES}
  )

TRIBITS_SUBPACKAGE_POSTPROCESS()
